function validar(codigoPostal, precio) {
    // Validación del código postal
    if (!/^[0-9]{5}$/.test(codigoPostal)) {
      console.log("El código postal debe ser un número de 5 dígitos.");
      return false;
    }
  
    // Validación del precio del artículo
    if (isNaN(precio) || precio <= 0) {
      console.log("El precio del artículo debe ser un número mayor que cero.");
      return false;
    }
  
    return true;
  }
  
  // Ejemplo de uso
  var codigoPostal = "12345";
  var precio = "19.99";
  if (validar(codigoPostal, precio)) {
    console.log("¡Campos válidos! Se puede proceder con el envío del formulario.");
  }
